import useBaseUrl from '@docusaurus/useBaseUrl';
import { books } from '@site/src/data/videos';
import styles from './videos.module.css';

export function VideoCard({ range, title, href }) {
  const thumbnailSrc = useBaseUrl(`/img/${range}-youtube.webp`);

  return (
    <article className={styles.videoCard}>
      <a className={styles.thumbnailLink} href={href}>
        <img className={styles.thumbnail} src={thumbnailSrc} alt={title} loading="lazy" />
      </a>

      <div className={styles.videoMeta}>
        <p className={styles.videoRange}>{range}</p>
        <p className={styles.videoTitle}>{title}</p>
      </div>
    </article>
  );
}

export function VideoLibrary() {
  return books.flatMap((book) => [
    <h2 key={`${book.title}-heading`}>{book.title}</h2>,
    ...book.sections.flatMap((section) => [
      section.title ? <h3 key={`${book.title}-${section.title}-heading`}>{section.title}</h3> : null,
      <div key={`${book.title}-${section.title || 'videos'}-grid`} className={styles.videoGrid}>
        {section.videos.map((video) => (
          <VideoCard key={video.range} {...video} />
        ))}
      </div>,
    ]).filter(Boolean),
  ]);
}
